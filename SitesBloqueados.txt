# Exemplos de como bloquear:

# Bloquear todos os subdomínios de um domínio (lembre-se de NÃO INSERIR a # do início da linha)
# ||facebook.com^$important

# Bloquear todos os subdomínios e TLDs de um domínio (lembre-se de NÃO INSERIR a # do início da linha)
# ||facebook.*^$important

# Bloquear um domínio específico (lembre-se de NÃO INSERIR a # do início da linha)
# ||facebook.com$important

# Bloquear qualquer domínio que contenha um termo específico
# /facebook/$important
